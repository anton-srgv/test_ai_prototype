const { test, expect } = require('@playwright/test');
const { testit } = require('testit-adapter-playwright');

test('Авторизация в Test IT', { tag: ['@regression'] }, async ({ page }) => {
  testit.workItemIds(['596']);
  await page.goto('/');

  // Step 1: Авторизация
  await page.fill('#username', process.env.LOGIN);
  await page.fill('#password', process.env.PASSWORD);
  await page.click('button:has-text("Login")');
  await expect(page.locator('.main-page')).toBeVisible();

  // Step 2: Изменение имени
  await page.fill('[name="name"]', 'Test Project Alpha');
  await page.click('button:has-text("Save")');

  // Step 3: Проверка сохранения
  await expect(page.locator('.notification')).toContainText('Сохранено');
});
