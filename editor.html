const { test, expect } = require('@playwright/test');
const { testit } = require('testit-adapter-playwright');
/**
 * @externalId 596
 * @recipeId 019cf6e9-2298-7660-b98d-167abdd032d5
 * @title Авторизация в Test IT
 * @description Кейс проверки авторизации в Test IT
 */
test('Авторизация в Test IT', { tag: ['@regression'] }, async ({ page }) => {
  testit.workItemIds(['596']);

  await page.goto('/');

  // Step 1: Перейти на страницу "Проекты"
  await page.click('a[href="/projects"]');
  await expect(page).toHaveURL(/projects/);

  // Step 2: Нажать на кнопку создания проекта
  await page.click('button:has-text("Создать проект")');
  await expect(page.locator('.modal')).toBeVisible();

  // Step 3: Заполнить все поля
  await page.fill('[name="projectName"]', 'Test Project');
  await page.fill('[name="projectCode"]', 'TP');

  // Step 4: Нажать кнопку "Сохранить"
  await page.click('button:has-text("Сохранить")');
  await expect(page.locator('.modal')).not.toBeVisible();

  // Step 5: Перейти на страницу авторизации
  await page.goto('/login');

  // Step 6: Ввести %Login
  await page.fill('#username', process.env.LOGIN);

  // Step 7: Ввести %Password
  await page.fill('#password', process.env.PASSWORD);

  // Step 8: Нажать кнопку Войти
  await page.click('button:has-text("Войти")');
  await expect(page.locator('.project-areas')).toBeVisible();

  // Step 9: Удалить локального пользователя
  await page.request.delete('/api/users/local');
});
